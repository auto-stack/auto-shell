// ~/.ashrc — ash startup script (like .bashrc)
//
// This file runs every time you start an interactive ash session. Use it to
// define your own functions (AutoLang `fn`), aliases, and settings.
//
// == Organizing functions across files ==
// To keep functions in separate files, `source` them here. For example,
// put work-related functions in ~/.config/ash/work.at and add a line:
//
//     source ~/.config/ash/work.at
//
// == Calling your functions ==
// Functions defined here are available at the prompt:
//     ash> greet("world")
//     ash> countdown(3)
//
// Edit this file and restart ash (or run `source ~/.ashrc`) to apply changes.

// ── greet — say hello to someone ──
fn greet(name) {
    print("hello, " + name + "!")
}

// ── countdown — count down from n to 1 ──
fn countdown(n) {
    var i = n
    while (i > 0) {
        print(i)
        i = i - 1
    }
    print("go!")
}

// ── banner — print a decorated banner line ──
fn banner(text) {
    print("=== " + text + " ===")
}

// ── host_info — show the machine name via system() ──
fn host_info() {
    print(system("hostname"))
}
